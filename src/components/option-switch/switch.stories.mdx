# Option Switch
An option switch allows the user to *one* of multiple option, and causes an immediate reaction. It can be used, for example, as a tab bar or to choose between a light and dark color scheme.

## Icons

If Icons are used, set their size to **16px**.

## Code Snippet

```tsx
import { Card, CardHeading, OptionSwitch, SwitchOption } from "@ocean-ui/ocean";
import { useState } from "react";

function MyComponent(args: any) {
  const [selected, setSelected] = useState('light');

  return (
    <Card
      width={500}
    >
      <CardHeading>Appearance Settings</CardHeading>
      <OptionSwitch label="Color Theme">
        <SwitchOption selected={selected === 'light'} onClick={() => setSelected('light')}>
          <UilSun size="16"/>
        </SwitchOption>
        <SwitchOption selected={selected === 'dark'} onClick={() => setSelected('dark')}>
          <UilMoon size="16"/>
        </SwitchOption>
      </OptionSwitch>
    </Card>
  );
};
```
